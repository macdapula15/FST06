mkdir [directory_name]

# Remove directory
rmdir [directory_name]

# Print/Present working directory
pwd

# List command
ls

# Change directory
cd [directory_path]

# Go back one directory
cd ..

# Create files
touch [file_name.extension]

# Remove files
rm [file_name.extension]
